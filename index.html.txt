  <div class="container">
    <h1>📅 Расписание занятий</h1>
    <select id="groupSelect" disabled><option>Загрузка...</option></select>
    <div id="content" class="loading">Загрузка расписания...</div>
  </div>

  <script>
    const GROUPS_KEY = 'lastGroup';
    const DAYS_ORDER = ['Понедельник', 'Вторник', 'Среда', 'Четверг', 'Пятница', 'Суббота'];

    async function init() {
      try {
        const res = await fetch('schedule.json');
        if (!res.ok) throw new Error('Файл schedule.json не найден или недоступен');
        const data = await res.json();
        renderSelect(data);
        const saved = localStorage.getItem(GROUPS_KEY);
        if (saved && data.some(g => g.group === saved)) {
          document.getElementById('groupSelect').value = saved;
          renderSchedule(data, saved);
        }
      } catch (e) {
        document.getElementById('content').innerHTML = `<div class="error">⚠️ Ошибка: ${e.message}<br>Убедись, что открываешь сайт через локальный сервер или GitHub Pages.</div>`;
        document.getElementById('groupSelect').disabled = true;
      }
    }

    function renderSelect(data) {
      const sel = document.getElementById('groupSelect');
      sel.innerHTML = '<option value="">Выберите группу</option>';
      data.sort((a,b) => a.group.localeCompare(b.group)).forEach(g => {
        sel.innerHTML += `<option value="${g.group}">${g.group}</option>`;
      });
      sel.disabled = false;
      sel.addEventListener('change', e => {
        localStorage.setItem(GROUPS_KEY, e.target.value);
        renderSchedule(data, e.target.value);
      });
    }

    function renderSchedule(data, group) {
      const container = document.getElementById('content');
      if (!group) return container.innerHTML = '<p class="empty">Выберите группу для просмотра</p>';
      const g = data.find(x => x.group === group);
      if (!g) return container.innerHTML = '<p class="empty">Группа не найдена</p>';

      const sortedDays = [...g.schedule].sort((a,b) => DAYS_ORDER.indexOf(a.day) - DAYS_ORDER.indexOf(b.day));
      const today = getTodayName();

      let html = '';
      sortedDays.forEach(d => {
        const isToday = d.day === today ? ' style="background:#f0f4ff;border-left:4px solid var(--primary)"' : '';
        html += `<div class="day-block"${isToday}>
          <h3 class="day-title">${d.day} ${d.day === today ? '(сегодня)' : ''}</h3>`;
        if (d.lessons.length === 0) {
          html += '<p class="empty">Нет занятий</p>';
        } else {
          html += `<table><thead><tr><th>Время</th><th>Предмет</th><th>Ауд.</th><th>Тип</th></tr></thead><tbody>
            ${d.lessons.map(l => `<tr>
              <td>${l.time}</td>
              <td>${l.subject}</td>
              <td>${l.room}</td>
              <td><span class="badge">${l.type || '—'}</span></td>
            </tr>`).join('')}</tbody></table>`;
        }
        html += '</div>';
      });
      container.innerHTML = html;
    }

    function getTodayName() {
      const days = ['Воскресенье', 'Понедельник', 'Вторник', 'Среда', 'Четверг', 'Пятница', 'Суббота'];
      return days[new Date().getDay()];
    }

    init();
  </script>
